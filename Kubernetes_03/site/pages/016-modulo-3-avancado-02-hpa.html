# ============================================================
# ARQUIVO: 02-hpa.yaml
# HPA - Horizontal Pod Autoscaler
#
# Imagine uma lanchonete:
# - Hora do almoï¿½o: fila enorme -> contratar mais atendentes
# - 15h da tarde: vazio -> dispensar atendentes
#
# HPA faz exatamente isso com pods!
# Monitora mï¿½tricas (CPU, memï¿½ria, custom) e:
# - Aumenta pods quando a carga sobe (scale out)
# - Diminui pods quando a carga cai (scale in)
#
# PRï¿½-REQUISITO: Metrics Server instalado
# Minikube: minikube addons enable metrics-server
# ============================================================

# Primeiro: o Deployment que o HPA vai gerenciar
apiVersion: apps/v1
kind: Deployment

metadata:
  name: app-para-escalar
  labels:
    app: escalavel

spec:
  replicas: 2               # HPA vai sobrescrever este valor!
  selector:
    matchLabels:
      app: escalavel

  template:
    metadata:
      labels:
        app: escalavel
    spec:
      containers:
        - name: app
          image: nginx:1.25
          ports:
            - containerPort: 80
          resources:
            requests:
              cpu: "100m"       # OBRIGATï¿½RIO para HPA por CPU!
              memory: "64Mi"    # HPA precisa saber o baseline
            limits:
              cpu: "300m"
              memory: "128Mi"

---
# ---------------------------------------------------------------
# O HPA
# ---------------------------------------------------------------

apiVersion: autoscaling/v2          # Use v2 para mï¿½tricas avanï¿½adas
kind: HorizontalPodAutoscaler

metadata:
  name: hpa-app-escalavel

spec:
  scaleTargetRef:                   # QUEM o HPA vai controlar?
    apiVersion: apps/v1
    kind: Deployment
    name: app-para-escalar          # Nome do Deployment acima

  minReplicas: 2                    # Mï¿½NIMO de pods (nunca abaixo!)
  maxReplicas: 10                   # Mï¿½XIMO de pods (nunca acima!)

  metrics:                          # QUAIS mï¿½tricas usar?

    # Mï¿½TRICA 1: CPU
    - type: Resource
      resource:
        name: cpu
        target:
          type: Utilization         # Porcentagem de uso
          averageUtilization: 70    # Se mï¿½dia > 70%: escala!
                                    # Se mï¿½dia < 70%: reduz!

    # Mï¿½TRICA 2: Memï¿½ria
    - type: Resource
      resource:
        name: memory
        target:
          type: AverageValue        # Valor mï¿½dio absoluto
          averageValue: 100Mi       # Se mï¿½dia > 100Mi: escala!

  # -------------------------------------------------------
  # COMPORTAMENTO DE ESCALA (avanï¿½ado)
  # -------------------------------------------------------
  behavior:
    scaleUp:                        # Como escalar PARA CIMA
      stabilizationWindowSeconds: 60  # Aguarda 60s antes de escalar
                                      # Evita escalar por pico momentï¿½neo
      policies:
        - type: Pods
          value: 2                  # Adiciona mï¿½ximo 2 pods por vez
          periodSeconds: 60         # A cada 60 segundos

    scaleDown:                      # Como reduzir pods
      stabilizationWindowSeconds: 300  # Aguarda 5 min antes de reduzir
                                       # Evita ficar subindo e descendo
      policies:
        - type: Pods
          value: 1                  # Remove mï¿½ximo 1 pod por vez
          periodSeconds: 120        # A cada 2 minutos

# ENTREVISTA Sï¿½NIOR:
# "HPA vs VPA vs KEDA - quando usar cada um?"
# RESPOSTA:
# HPA: Escala HORIZONTAL (mais pods). Bom para stateless.
# VPA: Escala VERTICAL (mais CPU/RAM por pod). Bom para DBs.
# KEDA: Escala por mï¿½tricas CUSTOMIZADAS (filas, eventos).
#       Ex: escala baseado em tamanho da fila SQS/Kafka.
