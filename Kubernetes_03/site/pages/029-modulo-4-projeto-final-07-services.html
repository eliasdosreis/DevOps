# -------------------------------------------------------
# SERVICE 1: Frontend
# Tipo: ClusterIP (s� tr�fego interno vindo do Ingress)

apiVersion: v1
kind: Service

metadata:
  name: frontend              # Ingress vai apontar para "frontend"
  namespace: ecommerce        # Mesmo namespace de tudo!

  labels:
    app: frontend
    componente: web

spec:
  type: ClusterIP             # Interno. Ingress faz a exposi��o externa.

  selector:
    app: frontend             # Encontra pods com label app=frontend

  ports:
    - name: http
      port: 80                # Service ouve na porta 80
      targetPort: 80          # Manda para porta 80 do container

---
# -------------------------------------------------------
# SERVICE 2: API Backend
# Tipo: ClusterIP (chamada do Frontend via Ingress /api)
# -------------------------------------------------------
apiVersion: v1
kind: Service

metadata:
  name: api                   # Frontend chama: http://api:3000
  namespace: ecommerce

  labels:
    app: api
    componente: backend

spec:
  type: ClusterIP

  selector:
    app: api                  # Pods do deployment "api"

  ports:
    - name: http
      port: 3000              # Porta da API Node.js
      targetPort: 3000

---
# -------------------------------------------------------
# SERVICE 3: PostgreSQL
# Tipo: ClusterIP (apenas a API acessa. Nunca expose DB!)

    app: postgres
