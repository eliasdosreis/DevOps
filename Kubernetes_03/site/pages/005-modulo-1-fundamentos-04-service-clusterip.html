# Primeiro criamos o Pod que o Service vai apontar

kind: Pod

  name: app-backend           # Pod que serï¿½ exposto pelo service
  labels:
    app: backend              # LABEL CRUCIAL! O Service usa isso
    versao: "v1"              # para encontrar o pod

  name: service-backend     # Nome do service
                            # Outros pods usam ESSE nome para

# ============================================================
# COMO FUNCIONA O SELECTOR:

#   Pod tem label:     app: backend  ?

#   RESULTADO: Service roteia trï¿½fego para o Pod!
