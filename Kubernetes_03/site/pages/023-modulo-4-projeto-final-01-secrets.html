# ============================================================
# ARQUIVO: 01-secrets.yaml
# PROJETO FINAL - Todos os Secrets centralizados
#
# REGRA DE OURO: NUNCA suba secrets reais para o git!
# Em produção use: Vault, AWS Secrets Manager, Sealed Secrets
#
# Como gerar os valores em base64:
#   echo -n "minha-senha" | base64
#
# TODOS os valores abaixo estão em BASE64:
# "postgres-senha-forte"  -> cG9zdGdyZXMtc2VuaGEtZm9ydGU=
# "redis-senha-forte"     -> cmVkaXMtc2VuaGEtZm9ydGU=
# "jwt-chave-secreta-256" -> and0LWNoYXZlLXNlY3JldGEtMjU2
# "admin"                 -> YWRtaW4=
# "ecommerce_db"          -> ZWNvbW1lcmNlX2Ri
# ============================================================

# SECRET 1: Credenciais do PostgreSQL
apiVersion: v1
kind: Secret

metadata:
  name: postgres-secret          # Nome do secret
  namespace: ecommerce           # SEMPRE especificar o namespace!
  labels:
    app: postgres
    componente: banco-de-dados

type: Opaque                     # Tipo genérico (mais comum)

data:                            # Valores em BASE64
  POSTGRES_USER: YWRtaW4=                      # "admin"
  POSTGRES_PASSWORD: cG9zdGdyZXMtc2VuaGEtZm9ydGU=  # "postgres-senha-forte"
  POSTGRES_DB: ZWNvbW1lcmNlX2Ri               # "ecommerce_db"
  # String de conexão completa (útil para a API)
  DATABASE_URL: cG9zdGdyZXNxbDovL2FkbWluOnBvc3RncmVzLXNlbmhhLWZvcnRlQHBvc3RncmVzLXNlcnZpY2U6NTQzMi9lY29tbWVyY2VfZGI=

---
# SECRET 2: Credenciais do Redis
apiVersion: v1
kind: Secret

metadata:
  name: redis-secret
  namespace: ecommerce
  labels:
    app: redis
    componente: cache

type: Opaque

data:
  REDIS_PASSWORD: cmVkaXMtc2VuaGEtZm9ydGU=    # "redis-senha-forte"

---
# SECRET 3: Chaves da API (JWT, etc.)
apiVersion: v1
kind: Secret

metadata:
  name: api-secret
  namespace: ecommerce
  labels:
    app: api
    componente: backend

type: Opaque

data:
  JWT_SECRET: and0LWNoYXZlLXNlY3JldGEtMjU2    # "jwt-chave-secreta-256"
  # Em produção: chave aleatória de 256+ bits
  # Gerar: openssl rand -base64 32 | base64

# ============================================================
# BOAS PRÁTICAS SÊNIOR DE SECRETS:
#
# 1. Separar secrets por componente (postgres, redis, api)
#    Princípio do mínimo acesso: pod do frontend não precisa
#    saber a senha do banco!
#
# 2. Em produção REAL, use External Secrets Operator:
#    Sincroniza do Vault/AWS Secrets Manager para K8s Secrets
#    O secret não fica no git NUNCA.
#
# 3. Habilite etcd encryption at rest no cluster
#
# VERIFICAR (sem mostrar valores):
#   kubectl get secrets -n ecommerce
#   kubectl describe secret postgres-secret -n ecommerce
# ============================================================
