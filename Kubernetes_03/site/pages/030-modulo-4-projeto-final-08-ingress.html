# PROJETO FINAL - Ingress com TLS e Roteamento
#
# Este ï¿½ o PORTEIRO PRINCIPAL da aplicaï¿½ï¿½o.
# Todo trï¿½fego da internet passa por aqui primeiro.
#
# Regras:
# loja.com/        --> Frontend (site da loja)
# loja.com/api/*   --> API Backend (dados dinï¿½micos)
#
# TLS (HTTPS) ï¿½ gerenciado pelo cert-manager (automï¿½tico!)
# ============================================================

apiVersion: networking.k8s.io/v1
kind: Ingress

metadata:
  name: ingress-ecommerce     # Nome do ingress
  namespace: ecommerce        # Mesmo namespace da app

  annotations:
    # --- Configuraï¿½ï¿½es do NGINX Ingress Controller ---

    kubernetes.io/ingress.class: "nginx"
    # Qual controller usar (pode ser traefik, kong, etc)

    nginx.ingress.kubernetes.io/ssl-redirect: "true"
    # Redireciona HTTP -> HTTPS automaticamente

    nginx.ingress.kubernetes.io/proxy-body-size: "10m"
    # Tamanho mï¿½ximo de upload: 10 megabytes
    # ï¿½til para upload de imagens de produtos

    nginx.ingress.kubernetes.io/proxy-read-timeout: "60"
    # Timeout para resposta do backend: 60 segundos

    nginx.ingress.kubernetes.io/rate-limit: "100"
    # Limite de requisiï¿½ï¿½es por segundo por IP
    # Proteï¿½ï¿½o bï¿½sica contra DDoS e bots

    nginx.ingress.kubernetes.io/enable-cors: "true"
    # Habilita CORS para o frontend poder chamar a API

    nginx.ingress.kubernetes.io/cors-allow-origin: "https://loja.com"
    # Sï¿½ permite CORS do nosso domï¿½nio

    # Se usar cert-manager para TLS automï¿½tico:
    # cert-manager.io/cluster-issuer: "letsencrypt-prod"

spec:
  # -------------------------------------------------------
  # TLS - Certificado HTTPS
  # -------------------------------------------------------
  tls:
    - hosts:
        - loja.com              # Domï¿½nio principal
        - www.loja.com          # Com www tambï¿½m
      secretName: tls-loja-com  # Secret com o certificado SSL
                                # cert-manager cria este Secret
                                # automaticamente!

  # -------------------------------------------------------
  # REGRAS DE ROTEAMENTO
  # -------------------------------------------------------
  rules:

    # --- Domï¿½nio principal: loja.com ---
    - host: loja.com
      http:
        paths:

          # ROTA 1: /api/* vai para o Backend
          - path: /api
            pathType: Prefix    # Qualquer coisa comeï¿½ando com /api
            backend:
              service:
                name: api       # Service da API (07-services.yaml)
                port:
                  number: 3000

          # ROTA 2: /* vai para o Frontend (sempre por ï¿½ltimo!)
          - path: /
            pathType: Prefix    # Captura tudo que nï¿½o foi acima
            backend:
              service:
                name: frontend  # Service do Frontend
                port:
                  number: 80

    # --- www.loja.com tambï¿½m funciona ---
    - host: www.loja.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: frontend
                port:
                  number: 80
