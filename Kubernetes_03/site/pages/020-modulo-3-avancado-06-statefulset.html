# ARQUIVO: 06-statefulset.yaml
# STATEFULSET - Para apps que tï¿½m identidade prï¿½pria
#
# Deployment cria pods ANï¿½NIMOS e INTERCAMBIï¿½VEIS.
# StatefulSet cria pods com IDENTIDADE FIXA e ORDEM.
#
# Analogia:
# Deployment = Soldados genï¿½ricos (qualquer um serve)
# StatefulSet = Membros de uma banda (cada um tem papel ï¿½nico)
#   - pod-0 = guitarrista (primï¿½rio/master)
#   - pod-1 = baixista   (rï¿½plica)
#   - pod-2 = baterista  (rï¿½plica)
#
# QUANDO USAR STATEFULSET?
# - Bancos de dados (MySQL, PostgreSQL, MongoDB)
# - Sistemas de cache com estado (Redis com persistence)
# - Message brokers (Kafka, RabbitMQ)
# - Qualquer app que precise de identidade de rede estï¿½vel
# ============================================================

apiVersion: apps/v1
kind: StatefulSet               # Tipo: StatefulSet

metadata:
  name: banco-mysql
  labels:
    app: mysql

spec:
  serviceName: "mysql-headless" # OBRIGATï¿½RIO! Nome do Headless Service
                                # Usado para DNS estï¿½vel dos pods

  replicas: 3                   # 3 instï¿½ncias do MySQL

  selector:
    matchLabels:
      app: mysql

  # -------------------------------------------------------
  # DIFERENï¿½A CHAVE: Pods tï¿½m nomes FIXOS e ORDENADOS
  # banco-mysql-0  (sempre o primeiro a iniciar = MASTER)
  # banco-mysql-1  (segundo)
  # banco-mysql-2  (terceiro)
  # -------------------------------------------------------

  template:
    metadata:
      labels:
        app: mysql

    spec:
      containers:
        - name: mysql
          image: mysql:8.0

          env:
            - name: MYSQL_ROOT_PASSWORD
              value: "senha-root"     # Use Secret em produï¿½ï¿½o!

          ports:
            - containerPort: 3306     # Porta padrï¿½o do MySQL

          volumeMounts:
            - name: dados-mysql       # Nome do volume (do VCT abaixo)
              mountPath: /var/lib/mysql  # Onde o MySQL guarda dados

  # -------------------------------------------------------
  # volumeClaimTemplates - EXCLUSIVO do StatefulSet!
  # Cria 1 PVC SEPARADO por pod automaticamente.
  # banco-mysql-0 -> pvc dados-mysql-banco-mysql-0
  # banco-mysql-1 -> pvc dados-mysql-banco-mysql-1
  # banco-mysql-2 -> pvc dados-mysql-banco-mysql-2
  # -------------------------------------------------------
  volumeClaimTemplates:
    - metadata:
        name: dados-mysql           # Prefixo do nome do PVC
      spec:
        accessModes: ["ReadWriteOnce"]
        resources:
          requests:
            storage: 10Gi           # 10GB por pod

---
# HEADLESS SERVICE - Obrigatï¿½rio para StatefulSet
# Headless = ClusterIP: None
# Permite acesso direto por DNS para cada pod:
# banco-mysql-0.mysql-headless.default.svc.cluster.local
# banco-mysql-1.mysql-headless.default.svc.cluster.local

apiVersion: v1
kind: Service

metadata:
  name: mysql-headless            # Nome referenciado no serviceName

spec:
  clusterIP: None                 # NONE = Headless Service!
                                  # Nï¿½o cria IP virtual, resolve
                                  # diretamente para IP do pod
  selector:
    app: mysql
  ports:
    - port: 3306
      targetPort: 3306

# ============================================================
# ENTREVISTA Sï¿½NIOR DIFï¿½CIL:
# "Qual a diferenï¿½a entre Deployment e StatefulSet?"
#
# DEPLOYMENT:
# - Pods intercambiï¿½veis (pod-xyz123, pod-abc456)
# - Sem ordem de inicializaï¿½ï¿½o garantida
# - Compartilham o mesmo PVC (ou nenhum)
# - Para apps STATELESS (APIs, frontends)
#
# STATEFULSET:
# - Pods com identidade fixa (app-0, app-1, app-2)
# - Inicia em ORDEM (0, depois 1, depois 2)
# - Cada pod tem seu PRï¿½PRIO PVC
# - DNS estï¿½vel: pod-0.service.namespace.svc.cluster.local
# - Para apps STATEFUL (bancos, brokers)
#
# "O que acontece quando vocï¿½ deleta um pod de StatefulSet?"
# RESPOSTA: O K8s recria com o MESMO nome e MESMO PVC.
# banco-mysql-1 morre -> nasce banco-mysql-1 com os mesmos dados!
