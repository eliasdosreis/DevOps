# ============================================================
# ARQUIVO: 09-hpa.yaml
# PROJETO FINAL - Auto Scaling da API
#
# A loja tem picos de acesso:
# - Black Friday: 10x mais usuï¿½rios!
# - Madrugada: quase ninguï¿½m
#
# HPA resolve isso automaticamente.
# Sem HPA: vocï¿½ paga por 10 pods 24h (mesmo sem usar)
# Com HPA: paga 2 pods normalmente, sobe para 10 no pico
# Economia real de dinheiro em cloud!
# ============================================================

apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler

metadata:
  name: hpa-api               # Nome do HPA
  namespace: ecommerce        # Mesmo namespace

spec:
  # O que o HPA vai controlar?
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: api                 # O Deployment da API (05-api-deployment.yaml)

  minReplicas: 2              # Mï¿½nimo: 2 pods SEMPRE
                              # Nunca menos (alta disponibilidade!)
  maxReplicas: 15             # Mï¿½ximo: 15 pods
                              # Limite de custo e capacidade do cluster

  # -------------------------------------------------------
  # Mï¿½TRICAS PARA DECISï¿½O DE SCALE
  # -------------------------------------------------------
  metrics:

    # Mï¿½trica 1: CPU mï¿½dia dos pods
    - type: Resource
      resource:
        name: cpu
        target:
          type: Utilization
          averageUtilization: 65   # Se CPU mï¿½dia > 65%: sobe pods
                                   # Se CPU mï¿½dia < 65%: desce pods
                                   # 65% ï¿½ um bom balanï¿½o (nï¿½o 80%!)

    # Mï¿½trica 2: Memï¿½ria mï¿½dia dos pods
    - type: Resource
      resource:
        name: memory
        target:
          type: AverageValue
          averageValue: 200Mi      # Se memï¿½ria mï¿½dia > 200MB: sobe

  # -------------------------------------------------------
  # COMPORTAMENTO FINO DO SCALING
  # -------------------------------------------------------
  behavior:

    # Como subir (scale out) - pode ser Rï¿½PIDO em picos
    scaleUp:
      stabilizationWindowSeconds: 30   # Espera 30s antes de subir
                                       # Evita scale por pico de 5 segundos
      policies:
        - type: Percent               # Escala por porcentagem
          value: 100                  # Pode DOBRAR os pods de uma vez
          periodSeconds: 60           # A cada 1 minuto
        - type: Pods                  # Ou adiciona pods fixos
          value: 3                    # Mï¿½ximo 3 pods por perï¿½odo
          periodSeconds: 60
      selectPolicy: Max               # Usa a polï¿½tica que escalar MAIS

    # Como descer (scale in) - deve ser CONSERVADOR
    scaleDown:
      stabilizationWindowSeconds: 300  # Espera 5 MINUTOS antes de descer
                                       # Evita flapping (sobe/desce/sobe)
      policies:
        - type: Pods
          value: 1                     # Remove sï¿½ 1 pod por vez
          periodSeconds: 120           # A cada 2 minutos
                                       # Descida lenta = mais segura!

# ============================================================
# TIMELINE DE UM PICO (BLACK FRIDAY):
#
# 23:59 - 2 pods rodando (normal)
# 00:00 - Black Friday comeï¿½a! CPU vai a 90%
# 00:00:30 - HPA detecta (stabilizationWindow=30s)
# 00:01:00 - Sobe para 4 pods (dobrou: policy Percent 100%)
# 00:01:30 - Ainda alta, sobe para 8 pods
# 00:02:00 - CPU se estabiliza em 60% com 8 pods
# ...
# 02:00 - Trï¿½fego cai, CPU vai a 20%
# 02:05 - HPA aguarda 5 min (stabilizationWindow=300s)
# 02:07 - Remove 1 pod -> 7 pods
# 02:09 - Remove 1 pod -> 6 pods  (lento e controlado)
# ...
# 04:00 - Volta para 2 pods (mï¿½nimo)
#
# ENTREVISTA Sï¿½NIOR:
# "Por que scaleDown ï¿½ mais lento que scaleUp?"
# RESPOSTA: Seguranï¿½a! Descer rï¿½pido demais pode causar
# sobrecarga nos pods restantes se o trï¿½fego voltar.
# Subir rï¿½pido ï¿½ seguro (mais pods = mais capacidade).
# Descer deve ser conservador para evitar instabilidade.
