# ============================================================
# ARQUIVO: 01-ingress.yaml
# O QUE É INGRESS?
#
# Imagine um grande shopping com uma recepção central:
# - Visitante chega: "Quero ir à loja de roupas"
# - Recepção: "Vá pelo corredor A, porta 3"
#
# Ingress é essa recepção! Recebe requisições HTTP/HTTPS
# e direciona para o Service correto baseado em:
# - Domínio: api.empresa.com -> Service A
# - Caminho: /api -> Service A, /web -> Service B
#
# Sem Ingress: precisa de 1 LoadBalancer por Service (caro!)
# Com Ingress: 1 LoadBalancer para TUDO (econômico!)
#
# PRÉ-REQUISITO: Ingress Controller instalado
# Minikube: minikube addons enable ingress
# ============================================================

# ------- Primeiro: os Services que o Ingress vai apontar -------

apiVersion: v1
kind: Service
metadata:
  name: service-frontend        # Service do frontend
spec:
  selector:
    app: frontend
  ports:
    - port: 80
      targetPort: 80

---

apiVersion: v1
kind: Service
metadata:
  name: service-api             # Service da API
spec:
  selector:
    app: api
  ports:
    - port: 8080
      targetPort: 8080

---
# ---------------------------------------------------------------
# AGORA O INGRESS
# ---------------------------------------------------------------

apiVersion: networking.k8s.io/v1    # API do Ingress
kind: Ingress                        # Tipo: Ingress

metadata:
  name: ingress-principal

  annotations:                       # Configurações do Ingress Controller
    nginx.ingress.kubernetes.io/rewrite-target: /
    # Remove o prefixo do path antes de enviar ao service
    # Ex: /api/users -> /users no container

    nginx.ingress.kubernetes.io/ssl-redirect: "true"
    # Força redirect HTTP -> HTTPS

spec:
  ingressClassName: nginx           # Qual Ingress Controller usar
                                    # Depende do que está instalado

  # -------------------------------------------------------
  # TLS - HTTPS (opcional mas recomendado em produção)
  # -------------------------------------------------------
  tls:
    - hosts:
        - meusite.com               # Domínio com HTTPS
        - api.meusite.com
      secretName: tls-certificado   # Secret com cert e chave SSL

  # -------------------------------------------------------
  # REGRAS DE ROTEAMENTO
  # -------------------------------------------------------
  rules:

    # Regra 1: tráfego para meusite.com
    - host: meusite.com             # Domínio (deve apontar para o cluster)
      http:
        paths:
          - path: /                 # Qualquer caminho /
            pathType: Prefix        # Prefix = começa com /
            backend:
              service:
                name: service-frontend    # Vai para este service
                port:
                  number: 80

    # Regra 2: tráfego para api.meusite.com
    - host: api.meusite.com         # Subdomínio diferente
      http:
        paths:
          - path: /v1               # Versão 1 da API
            pathType: Prefix
            backend:
              service:
                name: service-api
                port:
                  number: 8080

          - path: /health           # Health check da API
            pathType: Exact         # Exact = caminho exato
            backend:
              service:
                name: service-api
                port:
                  number: 8080

# ============================================================
# TIPOS DE pathType:
#
# Exact   --> Caminho exato. /api != /api/
# Prefix  --> Começa com. /api também pega /api/users
# ImplementationSpecific --> Depende do controller
#
# ENTREVISTA SÊNIOR:
# "Qual a diferença entre Ingress e Service tipo LoadBalancer?"
# RESPOSTA: LoadBalancer cria 1 IP externo por Service (caro).
# Ingress usa 1 LoadBalancer para rotear múltiplos Services
# baseado em host/path. Muito mais econômico em produção.
# ============================================================
