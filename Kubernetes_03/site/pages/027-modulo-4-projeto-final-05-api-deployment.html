# ============================================================
# ARQUIVO: 05-api-deployment.yaml
# PROJETO FINAL - API Backend (Node.js/Express)
#
# Este é o coração do nosso e-commerce.
# A API recebe requisições do frontend e:
# 1. Verifica autenticação (JWT)
# 2. Busca dados no PostgreSQL
# 3. Cacheia no Redis
# 4. Retorna resposta JSON
#
# RECURSOS USADOS AQUI:
# - Deployment com RollingUpdate
# - Liveness + Readiness + Startup Probes
# - Resources (requests/limits)
# - Variables de ConfigMap e Secret
# - SecurityContext (sem root)
# - Affinity (preferência de nó)
# ============================================================

apiVersion: apps/v1
kind: Deployment

metadata:
  name: api                      # Nome do deployment da API
  namespace: ecommerce

  annotations:
    kubernetes.io/change-cause: "API v1.0.0 - versao inicial"

  labels:
    app: api
    componente: backend
    versao: "1.0.0"

spec:
  replicas: 3                    # 3 réplicas para HA e load balancing

  selector:
    matchLabels:
      app: api

  # Zero downtime deploy
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1                # Máximo 1 pod extra durante update
      maxUnavailable: 0          # Nunca fica sem pods (zero downtime!)

  template:
    metadata:
      labels:
        app: api
        componente: backend
        versao: "1.0.0"

      # Anotações no pod (para Prometheus raspar métricas)
      annotations:
        prometheus.io/scrape: "true"    # Prometheus deve coletar métricas
        prometheus.io/port: "8080"      # De qual porta
        prometheus.io/path: "/metrics"  # Qual endpoint

    spec:
      # -------------------------------------------------------
      # securityContext: Segurança do pod
      # -------------------------------------------------------
      securityContext:
        runAsNonRoot: true         # NUNCA rodar como root!
        runAsUser: 1000            # UID do usuário da aplicação
        runAsGroup: 1000
        seccompProfile:
          type: RuntimeDefault     # Perfil de segurança padrão

      # -------------------------------------------------------
      # Terminar graciosamente (para zero downtime)
      # -------------------------------------------------------
      terminationGracePeriodSeconds: 30  # 30s para finalizar requisições

      # -------------------------------------------------------
      # affinity: Preferências de agendamento
      # Preferimos pods da API em nós DIFERENTES (HA!)
      # -------------------------------------------------------
      affinity:
        podAntiAffinity:           # Antiaffinidade = espalhar pods
          preferredDuringSchedulingIgnoredDuringExecution:
            - weight: 100          # Peso da preferência (0-100)
              podAffinityTerm:
                labelSelector:
                  matchLabels:
                    app: api       # Outros pods com app=api
                topologyKey: kubernetes.io/hostname  # Em nós diferentes

      containers:
        - name: api
          image: node:20-alpine   # Imagem Node.js (troque pela sua!)
                                  # Em produção: sua imagem customizada

          # Comando para iniciar a API
          command: ["node", "server.js"]

          ports:
            - containerPort: 8080
              name: http
              protocol: TCP

          # -------------------------------------------------------
          # ENV VARS: misturando ConfigMap e Secret
          # -------------------------------------------------------
          env:
            # Do ConfigMap geral
            - name: NODE_ENV
              valueFrom:
                configMapKeyRef:
                  name: config-geral
                  key: APP_ENV

            - name: DB_HOST
              valueFrom:
                configMapKeyRef:
                  name: config-geral
                  key: DB_HOST

            - name: DB_PORT
              valueFrom:
                configMapKeyRef:
                  name: config-geral
                  key: DB_PORT

            - name: REDIS_HOST
              valueFrom:
                configMapKeyRef:
                  name: config-geral
                  key: REDIS_HOST

            # Do ConfigMap da API
            - name: PORT
              valueFrom:
                configMapKeyRef:
                  name: config-api
                  key: PORT

            # Do Secret da API (dados sensíveis!)
            - name: JWT_SECRET
              valueFrom:
                secretKeyRef:
                  name: api-secret
                  key: JWT_SECRET

            # Da URL completa de conexão do banco
            - name: DATABASE_URL
              valueFrom:
                secretKeyRef:
                  name: postgres-secret
                  key: DATABASE_URL

            - name: REDIS_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: redis-secret
                  key: REDIS_PASSWORD

          # -------------------------------------------------------
          # Resources: API Node.js precisa de boa memória
          # -------------------------------------------------------
          resources:
            requests:
              memory: "256Mi"    # Node.js precisa de memória
              cpu: "200m"
            limits:
              memory: "512Mi"
              cpu: "1000m"       # Permite burst de CPU

          # -------------------------------------------------------
          # PROBES
          # -------------------------------------------------------

          # Startup: dá tempo para a API inicializar completamente
          startupProbe:
            httpGet:
              path: /health      # Endpoint simples de health
              port: 8080
            failureThreshold: 20  # 20 * 10s = 200s para iniciar
            periodSeconds: 10

          # Liveness: a API está viva e responsiva?
          livenessProbe:
            httpGet:
              path: /health
              port: 8080
            initialDelaySeconds: 15
            periodSeconds: 20
            timeoutSeconds: 5
            failureThreshold: 3

          # Readiness: a API está pronta para receber tráfego?
          # Verifica também se banco e redis estão conectados
          readinessProbe:
            httpGet:
              path: /ready       # Endpoint que verifica dependências!
              port: 8080
            initialDelaySeconds: 10
            periodSeconds: 10
            timeoutSeconds: 3
            failureThreshold: 3
            successThreshold: 1

          # securityContext no CONTAINER (complementa o do pod)
          securityContext:
            allowPrivilegeEscalation: false  # Não pode escalar privilégios
            readOnlyRootFilesystem: true     # Filesystem readonly (segurança)
            capabilities:
              drop: ["ALL"]                  # Remove TODAS as capabilities Linux

          # Volume para tmp (necessário com readOnlyRootFilesystem)
          volumeMounts:
            - name: tmp                      # Pasta /tmp writeable
              mountPath: /tmp

      volumes:
        - name: tmp
          emptyDir: {}                       # Volume temporário em memória

# ============================================================
# PERGUNTAS DIFÍCEIS DE ENTREVISTA:
#
# "O que é podAntiAffinity?"
# RESPOSTA: Regra que instrui o scheduler a EVITAR colocar
# pods com o mesmo label no mesmo nó. Garante que se um nó
# cair, não perdemos TODOS os pods da API.
#
# "Por que readOnlyRootFilesystem aumenta segurança?"
# RESPOSTA: Se um atacante comprometer o container, não
# consegue modificar o filesystem. Arquivos maliciosos
# não podem ser escritos. Reduz superfície de ataque.
#
# "O que é seccomp?"
# RESPOSTA: Filtra chamadas de sistema (syscalls) que o
# container pode fazer. RuntimeDefault bloqueia syscalls
# perigosas e raramente necessárias.
# ============================================================
