    const FILTERS = ["Todos", "Fundamentos", "Resilient", "High-Performing", "Secure", "Cost-Optimized"];
    let active = "Todos";
    const SK = "aws-saa-v4";
    let prog = JSON.parse(localStorage.getItem(SK) || "{}");

    function filtered() {
      return active === "Todos" ? mods : mods.filter((m) => m.domain === active);
    }

    function renderFilters() {
      const b = document.getElementById("filters");
      b.innerHTML = "";
      FILTERS.forEach((f) => {
        const el = document.createElement("button");
        el.type = "button";
        el.className = "chip" + (f === active ? " on" : "");
        el.textContent = f;
        el.addEventListener("click", () => {
          active = f;
          renderFilters();
          renderMods();
        });
        b.appendChild(el);
      });
    }

    function renderMods() {
      const grid = document.getElementById("moduleGrid");
      const pill = document.getElementById("countPill");
      const arr = filtered();
      pill.textContent = arr.length + " modulo" + (arr.length !== 1 ? "s" : "");
      grid.innerHTML = "";
      arr.forEach((m) => {
        const pct = DPCT[m.domain] || "-";
        const card = document.createElement("article");
        card.className = "mod-card";
        card.innerHTML = `
          <div class="mod-id" data-id="${m.id}">MODULO ${m.id}${pct !== "-" ? ` · ${pct}` : ""}</div>
          <h4 class="mod-title">${m.title}</h4>
          <p class="mod-desc">${m.desc}</p>
          <a class="open-link" href="${m.link}" target="_blank" rel="noopener noreferrer">Abrir sessao →</a>
        `;
        grid.appendChild(card);
      });

      grid.querySelectorAll(".mod-id[data-id]").forEach((el) => {
        el.addEventListener("click", (event) => {
          const id = event.currentTarget.getAttribute("data-id");
          prog[id] = !prog[id];
          localStorage.setItem(SK, JSON.stringify(prog));
          updateProg();
        });
      });
    }

    function updateProg() {
      const total = mods.length;
      const done = Object.values(prog).filter(Boolean).length;
      document.getElementById("progN").textContent = done;
      document.getElementById("progSub").textContent = `${done} concluido${done !== 1 ? "s" : ""}`;
      document.getElementById("progFill").style.width = `${Math.round((done / total) * 100)}%`;
    }

    function renderSims() {
      document.getElementById("simuladosGrid").innerHTML = sims
        .map(
          (s) => `
        <article class="sim-card">
          <h4 class="sim-title">${s.title}</h4>
          <p class="sim-desc">${s.desc}</p>
          ${s.href ? `<a class="open-link" href="${s.href}" target="_blank" rel="noopener noreferrer">Abrir sessao →</a>` : `<span class="open-link">${s.status}</span>`}
        </article>`
        )
        .join("");
    }

    renderSims();
    updateProg();
